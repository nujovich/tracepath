26/// Propagates name-value pairs in [W3C Baggage] format.
27///
28/// Baggage is used to annotate telemetry, adding context and
29/// information to metrics, traces, and logs. It is an abstract data type
30/// represented by a set of name-value pairs describing user-defined properties.
31/// Each name in a [`Baggage`] is associated with exactly one value.
32/// `Baggage`s are serialized according to the editor's draft of
33/// the [W3C Baggage] specification.
34///
35/// # Examples
36///
37/// ```
38/// use opentelemetry::{baggage::{Baggage, BaggageExt}, propagation::TextMapPropagator};
39/// use opentelemetry_sdk::propagation::BaggagePropagator;
40/// use std::collections::HashMap;
41///
42/// // Example baggage value passed in externally via http headers
43/// let mut headers = HashMap::new();
44/// headers.insert("baggage".to_string(), "user_id=1".to_string());
45///
46/// let propagator = BaggagePropagator::new();
47/// // can extract from any type that impls `Extractor`, usually an HTTP header map
48/// let cx = propagator.extract(&headers);
49///
50/// // Iterate over extracted name-value pairs
51/// for (name, value) in cx.baggage() {
52///     // ...
53/// }
54///
55/// // Add new baggage
56/// let mut baggage = Baggage::new();
57/// let _ = baggage.insert("server_id", "42");
58///
59/// let cx_with_additions = cx.with_baggage(baggage);
60///
61/// // Inject baggage into http request
62/// propagator.inject_context(&cx_with_additions, &mut headers);
63///
64/// let header_value = headers.get("baggage").expect("header is injected");
65/// assert!(!header_value.contains("user_id=1"), "still contains previous name-value");
66/// assert!(header_value.contains("server_id=42"), "does not contain new name-value pair");
67/// ```
68///
69/// [W3C Baggage]: https://w3c.github.io/baggage
70/// [`Baggage`]: opentelemetry::baggage::Baggage
71#[derive(Debug, Default)]
72pub struct BaggagePropagator {
73    _private: (),
74}

83impl TextMapPropagator for BaggagePropagator {
84    /// Encodes the values of the `Context` and injects them into the provided `Injector`.
85    fn inject_context(&self, cx: &Context, injector: &mut dyn Injector) {
86        let baggage = cx.baggage();
87        if !baggage.is_empty() {
88            let header_value = baggage
89                .iter()
90                .map(|(name, (value, metadata))| {
91                    let metadata_str = metadata.as_str().trim();
92                    let metadata_prefix = if metadata_str.is_empty() { "" } else { ";" };
93                    utf8_percent_encode(name.as_str().trim(), FRAGMENT)
94                        .chain(iter::once("="))
95                        .chain(utf8_percent_encode(value.as_str().trim(), FRAGMENT))
96                        .chain(iter::once(metadata_prefix))
97                        .chain(iter::once(metadata_str))
98                        .collect()
99                })
100                .collect::<Vec<String>>()
101                .join(",");
102            injector.set(BAGGAGE_HEADER, header_value);
103        }
104    }
105
106    /// Extracts a `Context` with baggage values from a `Extractor`.
107    fn extract_with_context(&self, cx: &Context, extractor: &dyn Extractor) -> Context {
108        if let Some(header_value) = extractor.get(BAGGAGE_HEADER) {
109            // Enforce the W3C Baggage maximum header length up-front so a
110            // single oversize header cannot drive per-entry allocation work
111            // before the entries are dropped on insert. See
112            // https://www.w3.org/TR/baggage/#limits.
113            if header_value.len() > MAX_BAGGAGE_LENGTH {
114                otel_warn!(
115                    name: "BaggagePropagator.Extract.HeaderTooLarge",
116                    message = "Baggage header exceeds W3C maximum length and was dropped",
117                    header_bytes = header_value.len() as i64,
118                    limit_bytes = MAX_BAGGAGE_LENGTH as i64,
119                );
120                return cx.clone();
121            }
122
123            let baggage =
124                header_value
125                    .split(',')
126                    .take(MAX_BAGGAGE_ITEMS)
127                    .filter_map(|context_value| {
128                        if let Some((name_and_value, props)) = context_value
129                            .split(';')
130                            .collect::<Vec<&str>>()
131                            .split_first()
132                        {
133                            let mut iter = name_and_value.split('=');
134                            if let (Some(name), Some(value)) = (iter.next(), iter.next()) {
135                                let decode_name = percent_decode_str(name).decode_utf8();
136                                let decode_value = percent_decode_str(value).decode_utf8();
137
138                                if let (Ok(name), Ok(value)) = (decode_name, decode_value) {
139                                    // Here we don't store the first ; into baggage since it should be treated
140                                    // as separator rather part of metadata
141                                    let decoded_props = props
142                                        .iter()
143                                        .flat_map(|prop| percent_decode_str(prop).decode_utf8())
144                                        .map(|prop| prop.trim().to_string())
145                                        .collect::<Vec<String>>()
146                                        .join(";"); // join with ; because we deleted all ; when calling split above
147
148                                    Some(KeyValueMetadata::new(
149                                        name.trim().to_owned(),
150                                        value.trim().to_string(),
151                                        decoded_props.as_str(),
152                                    ))
153                                } else {
154                                    otel_warn!(
155                                        name: "BaggagePropagator.Extract.InvalidUTF8",
156                                        message = "Invalid UTF8 string in key values",
157                                        baggage_header = header_value,
158                                    );
159                                    None
160                                }
161                            } else {
162                                otel_warn!(
163                                    name: "BaggagePropagator.Extract.InvalidKeyValueFormat",
164                                    message = "Invalid baggage key-value format",
165                                    baggage_header = header_value,
166                                );
167                                None
168                            }
169                        } else {
170                            otel_warn!(
171                        name: "BaggagePropagator.Extract.InvalidFormat",
172                        message = "Invalid baggage format",
173                        baggage_header = header_value);
174                            None
175                        }
176                    });
177            cx.with_baggage(baggage)
178        } else {
179            cx.clone()
180        }
181    }
182
183    fn fields(&self) -> FieldIter<'_> {
184        FieldIter::new(baggage_fields())
185    }
186}

347    #[test]
348    fn extract_drops_header_exceeding_max_length() {
349        let propagator = BaggagePropagator::new();
350
351        // Build a syntactically valid header longer than the W3C 8192-byte
352        // limit. The header must be rejected outright; no entries are added.
353        let mut header = String::with_capacity(MAX_BAGGAGE_LENGTH + 1024);
354        let mut i = 0u32;
355        while header.len() < MAX_BAGGAGE_LENGTH + 256 {
356            if !header.is_empty() {
357                header.push(',');
358            }
359            header.push_str(&format!("k{i}=v{i}"));
360            i += 1;
361        }
362        assert!(header.len() > MAX_BAGGAGE_LENGTH);
363
364        let mut extractor: HashMap<String, String> = HashMap::new();
365        extractor.insert(BAGGAGE_HEADER.to_string(), header);
366
367        let context = propagator.extract(&extractor);
368        assert_eq!(context.baggage().len(), 0);
369    }

371    #[test]
372    fn extract_caps_entry_count_at_max_baggage_items() {
373        let propagator = BaggagePropagator::new();
374
375        // Build a header whose total size stays under MAX_BAGGAGE_LENGTH but
376        // contains more than MAX_BAGGAGE_ITEMS entries.
377        let mut header = String::new();
378        let entries = MAX_BAGGAGE_ITEMS + 32;
379        for i in 0..entries {
380            if i > 0 {
381                header.push(',');
382            }
383            header.push_str(&format!("k{i:03}=v"));
384        }
385        assert!(header.len() <= MAX_BAGGAGE_LENGTH);
386
387        let mut extractor: HashMap<String, String> = HashMap::new();
388        extractor.insert(BAGGAGE_HEADER.to_string(), header);
389
390        let context = propagator.extract(&extractor);
391        // The storage type also caps at MAX_KEY_VALUE_PAIRS=64; the propagator
392        // additionally stops parsing at MAX_BAGGAGE_ITEMS so allocator work
393        // does not grow with attacker-supplied entry counts.
394        assert_eq!(context.baggage().len(), MAX_BAGGAGE_ITEMS);
395    }
