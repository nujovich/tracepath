36/// Builder for creating a new [LogExporter].
37#[derive(Debug, Default, Clone)]
38pub struct LogExporterBuilder<C> {
39    client: C,
40    endpoint: Option<String>,
41}
42
43impl LogExporterBuilder<NoExporterBuilderSet> {
44    /// Create a new [LogExporterBuilder] with default settings.
45    pub fn new() -> Self {
46        LogExporterBuilder::default()
47    }
48
49    /// With the gRPC Tonic transport.
50    #[cfg(feature = "grpc-tonic")]
51    pub fn with_tonic(self) -> LogExporterBuilder<TonicExporterBuilderSet> {
52        LogExporterBuilder {
53            client: TonicExporterBuilderSet(TonicExporterBuilder::default()),
54            endpoint: self.endpoint,
55        }
56    }
57
58    /// With the HTTP transport.
59    #[cfg(any(feature = "http-proto", feature = "http-json"))]
60    pub fn with_http(self) -> LogExporterBuilder<HttpExporterBuilderSet> {
61        LogExporterBuilder {
62            client: HttpExporterBuilderSet(HttpExporterBuilder::default()),
63            endpoint: self.endpoint,
64        }
65    }
66
67    /// Build the [LogExporter] with the default transport selected by environment
68    /// variable or feature flags.
69    ///
70    /// The transport is chosen based on:
71    /// 1. `OTEL_EXPORTER_OTLP_LOGS_PROTOCOL` environment variable
72    /// 2. `OTEL_EXPORTER_OTLP_PROTOCOL` environment variable
73    /// 3. Enabled features, with priority: `http-json` > `http-proto` > `grpc-tonic`
74    ///
75    /// Use [`with_tonic`](Self::with_tonic) or [`with_http`](Self::with_http) to
76    /// explicitly select a transport and access transport-specific configuration.
77    #[cfg(any(feature = "grpc-tonic", feature = "http-proto", feature = "http-json"))]
78    pub fn build(self) -> Result<LogExporter, ExporterBuildError> {
79        // NOTE: The transport-specific builder will call resolve_protocol again
80        // internally (for HTTP sub-protocol selection or tonic validation), but
81        // that's harmless — the result is the same.
82        let protocol = crate::exporter::resolve_protocol(OTEL_EXPORTER_OTLP_LOGS_PROTOCOL, None);
83        match protocol {
84            #[cfg(feature = "grpc-tonic")]
85            crate::Protocol::Grpc => self.with_tonic().build(),
86            #[cfg(feature = "http-proto")]
87            crate::Protocol::HttpBinary => self.with_http().build(),
88            #[cfg(feature = "http-json")]
89            crate::Protocol::HttpJson => self.with_http().build(),
90        }
91    }
92}

140/// OTLP exporter that sends log data
141#[derive(Debug)]
142pub struct LogExporter {
143    client: SupportedTransportClient,
144}
145
146#[derive(Debug)]
147enum SupportedTransportClient {
148    #[cfg(feature = "grpc-tonic")]
149    Tonic(crate::exporter::tonic::logs::TonicLogsClient),
150    #[cfg(any(feature = "http-proto", feature = "http-json"))]
151    Http(crate::exporter::http::OtlpHttpClient),
152}
153
154impl LogExporter {
155    /// Obtain a builder to configure a [LogExporter].
156    pub fn builder() -> LogExporterBuilder<NoExporterBuilderSet> {
157        LogExporterBuilder::default()
158    }
159
160    #[cfg(any(feature = "http-proto", feature = "http-json"))]
161    pub(crate) fn from_http(client: crate::exporter::http::OtlpHttpClient) -> Self {
162        LogExporter {
163            client: SupportedTransportClient::Http(client),
164        }
165    }
166
167    #[cfg(feature = "grpc-tonic")]
168    pub(crate) fn from_tonic(client: crate::exporter::tonic::logs::TonicLogsClient) -> Self {
169        LogExporter {
170            client: SupportedTransportClient::Tonic(client),
171        }
172    }
173}

175impl opentelemetry_sdk::logs::LogExporter for LogExporter {
176    async fn export(&self, batch: LogBatch<'_>) -> OTelSdkResult {
177        match &self.client {
178            #[cfg(feature = "grpc-tonic")]
179            SupportedTransportClient::Tonic(client) => client.export(batch).await,
180            #[cfg(any(feature = "http-proto", feature = "http-json"))]
181            SupportedTransportClient::Http(client) => client.export(batch).await,
182        }
183    }
184
185    fn set_resource(&mut self, resource: &opentelemetry_sdk::Resource) {
186        match &mut self.client {
187            #[cfg(feature = "grpc-tonic")]
188            SupportedTransportClient::Tonic(client) => client.set_resource(resource),
189            #[cfg(any(feature = "http-proto", feature = "http-json"))]
190            SupportedTransportClient::Http(client) => client.set_resource(resource),
191        }
192    }
193
194    fn shutdown_with_timeout(&self, _timeout: time::Duration) -> OTelSdkResult {
195        match &self.client {
196            #[cfg(feature = "grpc-tonic")]
197            SupportedTransportClient::Tonic(client) => client.shutdown(),
198            #[cfg(any(feature = "http-proto", feature = "http-json"))]
199            SupportedTransportClient::Http(client) => client.shutdown(),
200        }
201    }
202}
